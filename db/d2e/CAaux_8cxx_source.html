54 bool Track::GetPhiZat(float r, float bfield, float& phi, float& z) const
55 {
56  float rp4 = getCurvature(bfield);
57 
58  float xt = getY(), yt = getY();
59  float x = 0.f, y = 0.f;
60  float sn = sin(getAlpha()), cs = cos(getAlpha());
61  float a = x * cs + y * sn;
62  y = -x * sn + y * cs;
63  x = a;
64  xt -= x;
65  yt -= y;
66 
67  sn = rp4 * xt - getSnp();
68  cs = rp4 * yt + sqrt((1. - getSnp()) * (1. + getSnp()));
69  a = 2 * (xt * getSnp() - yt * sqrt((1. - getSnp()) * (1. + getSnp()))) - rp4 * (xt * xt + yt * yt);
70  float d = -a / (1 + sqrt(sn * sn + cs * cs));
71 
72  if (fabs(d) > r) {
73  if (r > 1e-1)
74  return false;
75  r = fabs(d);
76  }
77 
78  float rcurr = sqrt(getX() * getX() + getY() * getY());
79  float phicurr = getPhi();
80 
81  if (getX() >= 0.) {
82  phi = phicurr + asin(d / r) - asin(d / rcurr);
83  } else {
84  phi = phicurr + asin(d / r) + asin(d / rcurr) - PI;
85  }
86 
87  // return a phi in [0,2pi
88  if (phi < 0.)
89  phi += 2. * PI;
90  else if (phi >= 2. * PI)
91  phi -= 2. * PI;
92  z = getZ() + getTgl() * (sqrt((r - d) * (r + d)) - sqrt((rcurr - d) * (rcurr + d)));
93 
94  return true;
95 }
