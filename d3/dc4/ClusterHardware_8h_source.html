30  uint16_t qMax; //< QMax of the cluster

35  float getPad() const { return padPre / qTot; }

39  } // Returns the local time, not taking into accound the time bin offset of the container
40  float getSigmaPad2() const { return (sigmaPad2Pre - padPre * padPre) / (qTot * qTot); }
