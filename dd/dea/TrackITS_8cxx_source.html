41 void TrackITS::getImpactParams(Float_t x, Float_t y, Float_t z, Float_t bz, Float_t ip[2]) const
42 {
43  //------------------------------------------------------------------
44  // This function calculates the transverse and longitudinal impact parameters
45  // with respect to a point with global coordinates (x,y,0)
46  // in the magnetic field "bz" (kG)
47  //------------------------------------------------------------------
48  Float_t f1 = getSnp(), r1 = TMath::Sqrt((1. - f1) * (1. + f1));
49  Float_t xt = getX(), yt = getY();
50  Float_t sn = TMath::Sin(getAlpha()), cs = TMath::Cos(getAlpha());
51  Float_t a = x * cs + y * sn;
52  y = -x * sn + y * cs;
53  x = a;
54  xt -= x;
55  yt -= y;
56 
57  Float_t rp4 = getCurvature(bz);
58  if ((TMath::Abs(bz) < Almost0) || (TMath::Abs(rp4) < Almost0)) {
59  ip[0] = -(xt * f1 - yt * r1);
60  ip[1] = getZ() + (ip[0] * f1 - xt) / r1 * getTgl() - z;
61  return;
62  }
63 
64  sn = rp4 * xt - f1;
65  cs = rp4 * yt + r1;
66  a = 2 * (xt * f1 - yt * r1) - rp4 * (xt * xt + yt * yt);
67  Float_t rr = TMath::Sqrt(sn * sn + cs * cs);
68  ip[0] = -a / (1 + rr);
69  Float_t f2 = -sn / rr, r2 = TMath::Sqrt((1. - f2) * (1. + f2));
70  ip[1] = getZ() + getTgl() / rp4 * TMath::ASin(f2 * r1 - f1 * r2) - z;
71 }
