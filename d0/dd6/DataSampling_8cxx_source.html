39 {

55 auto DataSampling::getEdgeCreator(const QcTaskConfiguration& taskCfg, const InfrastructureConfig& infrastructureCfg)
56 {
57  return taskCfg.fairMqOutputChannelConfig.empty() ?
58  (infrastructureCfg.enableParallelDispatchers ?
59  [](DataProcessorSpec& dispatcher, const InputSpec& newInput) {
60  dispatcher.inputs.push_back(newInput);
61  dispatcher.outputs.push_back(createDispatcherOutputSpec(newInput));
62  }
63  :
64  [](DataProcessorSpec& dispatcher, const InputSpec& newInput) {
65  dispatcher.inputs.push_back(newInput);
66  OutputSpec newOutput = createDispatcherOutputSpec(newInput);
67  if (std::find(dispatcher.outputs.begin(), dispatcher.outputs.end(), newOutput)
68  == dispatcher.outputs.end()) {
69  dispatcher.outputs.push_back(newOutput);
70  }
71  })
72  :
73  [](DataProcessorSpec& dispatcher, const InputSpec& newInput) {
74  dispatcher.inputs.push_back(newInput);
75  };
76 }

80 auto DataSampling::getDispatcherCreator(const QcTaskConfiguration& taskCfg)
81 {
82  return taskCfg.fairMqOutputChannelConfig.empty() ?
83  // create dispatcher with DPL output
84  [](const SubSpecificationType& dispatcherSubSpec, const QcTaskConfiguration& task, const InputSpec& input) {
85  return DataProcessorSpec{
86  "Dispatcher" + std::to_string(dispatcherSubSpec) + "_for_" + task.name,
87  Inputs{input},
88  Outputs{createDispatcherOutputSpec(input)},
89  AlgorithmSpec{
90  [gen = BernoulliGenerator(task.fractionOfDataToSample)](ProcessingContext& ctx) mutable {
91  DataSampling::dispatcherCallback(ctx, gen);
92  }
93  }
94  };
95  }
96  :
97  // create dispatcher with FairMQ output
98  [](const SubSpecificationType& dispatcherSubSpec, const QcTaskConfiguration& task, const InputSpec& newInput) {
99  //todo: throw an exception when 'name=' not found?
100  size_t n_begin = task.fairMqOutputChannelConfig.find("name=") + sizeof("name=") - 1;
101  size_t n_end = task.fairMqOutputChannelConfig.find_first_of(',', n_begin);
102  std::string channel = task.fairMqOutputChannelConfig.substr(n_begin, n_end - n_begin);
103 
104  return DataProcessorSpec{
105  "Dispatcher" + std::to_string(dispatcherSubSpec) + "_for_" + task.name,
106  Inputs{newInput},
107  Outputs{},
108  AlgorithmSpec{
109  [fraction = task.fractionOfDataToSample, channel](InitContext& ctx) {
110  return dispatcherInitCallbackFairMQ(ctx, channel, fraction);
111  }
112  },
113  {
114  ConfigParamSpec{
115  "channel-config", VariantType::String,
116  task.fairMqOutputChannelConfig.c_str(), {"Out-of-band channel config"}}
117  }
118  };
119  };
120 }

122 void DataSampling::GenerateInfrastructure(WorkflowSpec& workflow, const std::string& configurationSource)
123 {
124  QcTaskConfigurations tasks = readQcTasksConfiguration(configurationSource);
125  InfrastructureConfig infrastructureCfg = readInfrastructureConfiguration(configurationSource);
126 
127  for (auto&& task : tasks) {
128 
129  // if necessary, create FairMQ -> DPL proxies and after that, look for their outputs in workflow.
130  // (it is surely not an optimal way to do that, but this is only a temporary feature)
131  for (auto&& fairMqProxy : task.desiredFairMqData) {
132  workflow.emplace_back(specifyExternalFairMQDeviceProxy(
133  ("FairMQ_proxy_for_" + task.name).c_str(),
134  Outputs{fairMqProxy.outputSpec},
135  fairMqProxy.channelConfig.c_str(),
136  fairMqProxy.converterType == "o2DataModelAdaptor" ? o2DataModelAdaptor(fairMqProxy.outputSpec, 0, 1)
137  : incrementalConverter(fairMqProxy.outputSpec, 0, 1)
138  ));
139  }
140 
141  std::unordered_map<SubSpecificationType, DataProcessorSpec> dispatchers;
142 
143  // some lambda functions to make the later code cleaner and hide its configuration
144  auto areEdgesMatching = getEdgeMatcher(task);
145  auto createDispatcherSpec = getDispatcherCreator(task);
146  auto addEdgesToDispatcher = getEdgeCreator(task, infrastructureCfg);
147 
148  // Find all available outputs in workflow that match desired data. Create dispatchers that take them as inputs
149  // and provide them filtered as outputs.
150  for (auto&& desiredData : task.desiredDataSpecs) {
151  bool wasDataFound = false;
152  for (auto&& dataProcessor : workflow) {
153  for (auto&& externalOutput : dataProcessor.outputs) {
154  if (areEdgesMatching(externalOutput, desiredData, task.subSpec)) {
155 
156  wasDataFound = true;
157  InputSpec newInput{
158  desiredData.binding,
159  desiredData.origin,
160  desiredData.description,
161  externalOutput.subSpec,
162  static_cast<InputSpec::Lifetime>(externalOutput.lifetime),
163  };
164 
165  // if parallel dispatchers are not enabled, then edges will be added to the only one dispatcher.
166  // in other case, new dispatcher will be created for every parallel flow.
167  SubSpecificationType dispatcherSubSpec = infrastructureCfg.enableParallelDispatchers ?
168  externalOutput.subSpec : 0;
169 
170  auto res = dispatchers.find(dispatcherSubSpec);
171  if (res != dispatchers.end()) {
172 
173  addEdgesToDispatcher(res->second, newInput);
174 
175  if (infrastructureCfg.enableTimePipeliningDispatchers &&
176  res->second.maxInputTimeslices < dataProcessor.maxInputTimeslices) {
177  res->second.maxInputTimeslices = dataProcessor.maxInputTimeslices;
178  }
179  } else {
180  dispatchers[dispatcherSubSpec] = createDispatcherSpec(dispatcherSubSpec, task, newInput);
181 
182  //todo: can fairmq dispatchers be time pipelined?
183  if (infrastructureCfg.enableTimePipeliningDispatchers) {
184  dispatchers[dispatcherSubSpec].maxInputTimeslices = dataProcessor.maxInputTimeslices;
185  }
186  }
187  }
188  }
189  }
190  if (!wasDataFound) {
191  LOG(ERROR) << "Data '" << desiredData.binding << "' for QC task '" << task.name
192  << "' not found in given workflow";
193  }
194  }
195  for (auto& dispatcher : dispatchers) {
196  workflow.push_back(std::move(dispatcher.second));
197  }
198  }
199 }

210 void DataSampling::dispatcherCallback(ProcessingContext& ctx, BernoulliGenerator& bernoulliGenerator)
211 {
212  InputRecord& inputs = ctx.inputs();
213 
214  if (bernoulliGenerator.drawLots()) {
215  for (auto& input : inputs) {
216 
217  OutputSpec outputSpec = createDispatcherOutputSpec(*input.spec);
218 
219  const auto* inputHeader = Header::get<Header::DataHeader>(input.header);
220 
221  if (inputHeader->payloadSerializationMethod == Header::gSerializationMethodInvalid) {
222  LOG(ERROR) << "DataSampling::dispatcherCallback: input of origin'" << inputHeader->dataOrigin.str
223  << "', description '" << inputHeader->dataDescription.str
224  << "' has gSerializationMethodInvalid.";
225  } else if (inputHeader->payloadSerializationMethod == Header::gSerializationMethodROOT) {
226  ctx.allocator().adopt(outputSpec, DataRefUtils::as<TObject>(input).release());
227  } else { //POD
228  //todo: use API for that when it is available
229  ctx.allocator().adoptChunk(outputSpec, const_cast<char*>(input.payload), inputHeader->size(),
230  &Header::Stack::freefn, nullptr);
231  }
232 
233  LOG(DEBUG) << "DataSampler sends data from subspec " << input.spec->subSpec;
234  }
235  }

249 void DataSampling::dispatcherCallbackFairMQ(ProcessingContext& ctx, BernoulliGenerator& bernoulliGenerator,
250  FairMQDevice* device, const std::string& channel)
251 {
252  InputRecord& inputs = ctx.inputs();
253 
254  if (bernoulliGenerator.drawLots()) {
255 
256  auto cleanupFcn = [](void* data, void* hint) { delete[] reinterpret_cast<char*>(data); };
257  for (auto& input : inputs) {
258  const auto* header = Header::get<header::DataHeader>(input.header);
259 
260  char* payloadCopy = new char[header->payloadSize];
261  memcpy(payloadCopy, input.payload, header->payloadSize);
262  FairMQMessagePtr msgPayload(device->NewMessage(payloadCopy, header->payloadSize, cleanupFcn, payloadCopy));
263 
264  int bytesSent = device->Send(msgPayload, channel);
265  LOG(DEBUG) << "Payload bytes sent: " << bytesSent;
266  }

307 DataSampling::QcTaskConfigurations DataSampling::readQcTasksConfiguration(const std::string& configurationSource)
308 {
309  std::vector<QcTaskConfiguration> tasks;
310  std::unique_ptr<ConfigurationInterface> configFile = ConfigurationFactory::getConfiguration(configurationSource);
311 
312  std::vector<std::string> taskNames;
313  try {
314  std::string taskNamesString = configFile->getString("DataSampling/tasksList").value();
315  boost::split(taskNames, taskNamesString, boost::is_any_of(","));
316  } catch (const boost::bad_optional_access&) {
317  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource
318  << ", wrong or missing value DataSampling/tasksList";
319  return QcTaskConfigurations();
320  }
321 
322  for (auto&& taskName : taskNames) {
323 
324  QcTaskConfiguration task;
325  task.name = taskName;
326 
327  std::string taskInputsNames;
328  try {
329  std::string simpleQcTaskDefinition = configFile->getString(taskName + "/taskDefinition").value();
330  taskInputsNames = configFile->getString(simpleQcTaskDefinition + "/inputs").value();
331  task.fractionOfDataToSample = configFile->getFloat(simpleQcTaskDefinition + "/fraction").value();
332  if (task.fractionOfDataToSample <= 0 || task.fractionOfDataToSample > 1) {
333  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource << ", value "
334  << simpleQcTaskDefinition + "/fraction" << " is not in range (0,1]. Setting value to 0.";
335  task.fractionOfDataToSample = 0;
336  }
337  //if there is a channelConfig specified, then user wants output in raw FairMQ layer, not DPL
338  task.fairMqOutputChannelConfig = configFile->getString(simpleQcTaskDefinition + "/channelConfig").value_or("");
339 
340  //FIXME: I do not like '-1' meaning 'all' - not 100% sure if it's safe to compare with '-1' later
341  task.subSpec = static_cast<header::DataHeader::SubSpecificationType>(
342  configFile->getInt(simpleQcTaskDefinition + "/subSpec").value_or(-1));
343 
344  } catch (const boost::bad_optional_access&) {
345  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource
346  << ", missing value or values for task " << taskName;
347  continue;
348  }
349 
350  std::vector<std::string> taskInputsSplit;
351  boost::split(taskInputsSplit, taskInputsNames, boost::is_any_of(","));
352 
353  for (auto&& input : taskInputsSplit) {
354 
355  InputSpec desiredData;
356  try {
357  desiredData.binding = configFile->getString(input + "/inputName").value();
358 
359  std::string origin = configFile->getString(input + "/dataOrigin").value();
360  origin.copy(desiredData.origin.str, (size_t) desiredData.origin.size);
361 
362  std::string description = configFile->getString(input + "/dataDescription").value();
363  description.copy(desiredData.description.str, (size_t) desiredData.description.size);
364 
365  } catch (const boost::bad_optional_access&) {
366  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource
367  << " input " << input << " has missing values";
368  continue;
369  }
370  task.desiredDataSpecs.push_back(desiredData);
371 
372  // for temporary feature
373  if (configFile->getInt(input + "/spawnConverter").value_or(0)) {
374  FairMqInput fairMqInput{
375  OutputSpec{
376  desiredData.origin,
377  desiredData.description,
378  task.subSpec == -1 ? 0 : task.subSpec,
379  },
380  configFile->getString(input + "/channelConfig").value_or(""),
381  configFile->getString(input + "/converterType").value_or("incrementalConverter")
382  };
383  task.desiredFairMqData.push_back(fairMqInput);
384  }
385  }
386 
387  if (task.desiredDataSpecs.empty()) {
388  LOG(ERROR) << "QC Task configuration error. In file " << configurationSource
389  << " task " << taskName << " has no valid inputs";
390  continue;
391  }
392  tasks.push_back(task);
393  }
394 
395  return tasks;
396 }
